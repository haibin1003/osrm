### 1.2 分层职责

| 层级 | 职责 | 技术组件 | 设计原则 |
|------|------|---------|---------|
| **展示层** | 用户界面渲染、交互逻辑 | Vue 3、Element Plus、ECharts | 组件化、响应式 |
| **网关层** | 请求路由、负载均衡、安全防护 | Nginx、Spring Cloud Gateway | 高可用、可扩展 |
| **应用层** | 业务逻辑编排、流程控制、权限校验 | Spring Boot、Spring Security | 事务控制、AOP |
| **领域层** | 核心业务逻辑、领域模型、业务规则 | DDD、领域服务 | 高内聚、低耦合 |
| **基础设施层** | 数据持久化、外部服务集成、通用工具 | JPA、Redis、MinIO | 可替换、可测试 |

### 3.2 实体关系说明

| 实体 | 类型 | 所属领域 | 说明 |
|------|------|---------|------|
| **User** | 聚合根 | 用户域 | 用户实体，管理用户生命周期 |
| **Role** | 实体 | 用户域 | 角色实体，与User多对多 |
| **SoftwarePackage** | 聚合根 | 软件域 | 软件包聚合根，管理版本集合 |
| **PackageVersion** | 实体 | 软件域 | 软件版本，生命周期状态管理 |
| **Attachment** | 值对象 | 软件域 | 附件信息，不可独立存在 |
| **ReviewRecord** | 实体 | 软件域 | 审核记录，记录审批历史 |
| **Subscription** | 聚合根 | 订购域 | 订购聚合根，管理订购流程 |
| **DownloadToken** | 值对象 | 订购域 | 下载令牌，包含有效期和次数 |
| **UsageRecord** | 实体 | 订购域 | 使用记录，追踪实际使用情况 |

```java
// 轮询任务：定期扫描未发送事件
@Component
public class OutboxPoller {

    @Scheduled(fixedRate = 5000) // 每5秒扫描一次
    public void pollAndSend() {
        List<OutboxRecord> pendingEvents = outboxRepository
            .findByStatusAndCreatedAtBefore(
                OutboxStatus.PENDING,
                Instant.now().minusSeconds(60)
            );

        for (OutboxRecord event : pendingEvents) {
            try {
                messageBroker.send(event.getTopic(), event.getPayload());
                event.markAsSent();
            } catch (Exception e) {
                event.incrementRetryCount();
                if (event.getRetryCount() > 3) {
                    event.markAsFailed();
                    alertService.sendAlert("事件发送失败", event);
                }
            }
        }
    }
}
```

#### 6.1.4 消费者幂等性设计

```java
// 幂等性保障：唯一键去重
@Component
public class IdempotentEventHandler {

    @RabbitListener(queues = "osrm.events.package")
    public void handlePackageEvent(Message message, Channel channel,
                                    @Header(AmqpHeaders.DELIVERY_TAG) long tag) {
        String eventId = message.getMessageProperties().getMessageId();

        // 1. 尝试插入消费记录（唯一键:eventId+consumerId）
        boolean isNew = consumptionLogRepository.tryInsert(eventId, consumerId);

        if (!isNew) {
            // 已消费过，直接确认消息
            channel.basicAck(tag, false);
            return;
        }

        try {
            // 2. 执行业务逻辑
            processEvent(message);

            // 3. 确认消息
            channel.basicAck(tag, false);
        } catch (Exception e) {
            // 4. 处理失败，删除消费记录（或标记为失败）
            consumptionLogRepository.delete(eventId, consumerId);
            channel.basicNack(tag, false, true); // 重新入队
        }
    }
}
```

```java
@Entity
@Table(name = "t_software_package")
public class SoftwarePackage {
    @Id
    private Long id;

    // 完整领域逻辑，包含验证、状态机转换等
    private String name;
    private PackageStatus status;

    public void submitForReview() {
        if (this.status != PackageStatus.DRAFT) {
            throw new IllegalStateException("只有草稿状态可以提交审核");
        }
        this.status = PackageStatus.PENDING;
        registerEvent(new PackageSubmittedEvent(this.id));
    }
}

// ========== 查询端模型（View Model）==========
// 专为查询优化，可使用@Subselect或直接查询视图
@Entity
@Subselect("""
    SELECT
        sp.id,
        sp.name,
        sp.status,
        sp.category_code,
        c.name as category_name,
        COUNT(DISTINCT s.id) as subscription_count,
        AVG(r.rating) as avg_rating
    FROM t_software_package sp
    LEFT JOIN t_category c ON sp.category_code = c.code
    LEFT JOIN t_subscription s ON s.package_id = sp.id AND s.status = 'ACTIVE'
    LEFT JOIN t_review r ON r.package_id = sp.id
    GROUP BY sp.id
""")
public class SoftwarePackageView {
    @Id
    private Long id;
    private String name;
    private String status;
    private String categoryName;
    private Long subscriptionCount;
    private Double avgRating;
}

@Transactional(readOnly = true)
public class SoftwarePackageQueryService {

    @Autowired
    private SoftwarePackageViewRepository viewRepository;

    // 复杂查询使用QueryDSL或原生SQL
    public Page<SoftwarePackageView> search(PackageSearchCriteria criteria) {
        return viewRepository.findAll(

                .name.containsIgnoreCase(criteria.getKeyword())
                .and(QSoftwarePackageView.softwarePackageView.status.eq("PUBLISHED")),
            criteria.toPageRequest()
        );
    }

    // 聚合查询：门户首页统计
    public PortalStatistics getPortalStatistics() {
```

#### 6.3.2 限流配置示例

```yaml
# Spring Cloud Gateway 限流配置
spring:
  cloud:
    gateway:
      routes:
        - id: osrm-api
          uri: lb://osrm-service
          predicates:
            - Path=/api/**
          filters:
            # 全局限流
            - name: RequestRateLimiter
              args:
                redis-rate-limiter.replenishRate: 1000
                redis-rate-limiter.burstCapacity: 2000
                key-resolver: "#{@userKeyResolver}"
            # 熔断配置
            - name: CircuitBreaker
              args:
                name: osrmCircuitBreaker
                fallbackUri: forward:/fallback

# 自定义限流key解析器（按用户限流）
@Component
public class RateLimitConfig {

    @Bean
    public KeyResolver userKeyResolver() {
        return exchange -> Mono.just(
            exchange.getRequest().getHeaders().getFirst("X-User-Id")
                != null ? exchange.getRequest().getHeaders().getFirst("X-User-Id")
                : exchange.getRequest().getRemoteAddress().getAddress().getHostAddress()
        );
    }
}
```

#### 6.3.3 熔断降级配置

```java
// Resilience4j 熔断配置
@Configuration
public class CircuitBreakerConfig {

    @Bean
    public Customizer<Resilience4JCircuitBreakerFactory> defaultCustomizer() {
        return factory -> factory.configureDefault(id -> new Resilience4JConfigBuilder(id)
            .circuitBreakerConfig(CircuitBreakerConfig.custom()
                .failureRateThreshold(60)          // 失败率阈值
                .waitDurationInOpenState(Duration.ofSeconds(30))  // 熔断后等待时间
                .permittedNumberOfCallsInHalfOpenState(5)  // 半开状态允许的请求数
                .slidingWindowSize(100)            // 滑动窗口大小
                .build())
            .timeLimiterConfig(TimeLimiterConfig.custom()
                .timeoutDuration(Duration.ofSeconds(5))  // 超时时间
                .build())
            .build());
    }
}

// 服务降级方法
@RestController
public class FallbackController {

    @RequestMapping("/fallback")
    public ResponseEntity<ApiResponse<?>> fallback() {
        return ResponseEntity.status(HttpStatus.SERVICE_UNAVAILABLE)
            .body(ApiResponse.error("服务暂时不可用，请稍后重试"));
    }
}
```
